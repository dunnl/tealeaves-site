Inductive K2 : Type := KType | KTerm.

#[export] Instance Keq : EqDec K2 eq.
Proof.
  change (forall x y : K2, {x = y} + {x <> y}).

Section operations.

  Context
    (F : Type -> Type)

  Fixpoint bind_type (f : forall (k : K), list K2 * A -> F (SystemF k B)) (t : typ A) : F (typ B) :=
    match t with
    | ty_c t =>
      pure F (ty_c t)
    | ty_v a =>
      f KType (nil, a)
    | ty_ar t1 t2 =>
      pure F (ty_ar) <â‹†> (bind_type f t1) <â‹†> (bind_type f t2)
    | ty_univ body =>
      pure F (ty_univ) <â‹†> (bind_type (fun k => preincr [KType] (f k)) body)
    end.

  Fixpoint bind_term (f : forall (k : K), list K2 * A -> F (SystemF k B)) (t : term A) : F (term B) :=
    match t with
    | tm_var a =>
      f KTerm (nil, a)
    | tm_abs ty body =>
      pure F (tm_abs)
           <â‹†> bind_type (fun k => f k) ty
           <â‹†> bind_term (fun k => f k âˆ˜ incr [KTerm]) body
    | tm_app t1 t2 =>
      pure F tm_app <â‹†> bind_term f t1 <â‹†> bind_term f t2
    | tm_tab body =>
      pure F tm_tab <â‹†> (bind_term (fun k => f k âˆ˜ incr [KType]) body)
    | tm_tap t1 ty =>
      pure F tm_tap <â‹†> bind_term f t1 <â‹†> bind_type f ty
    end.

End operations.
