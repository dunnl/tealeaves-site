#[export] Instance Fmap_option : Fmap option :=
  fun A B (f : A -> B) (m : option A) =>
    match m with
    | Some a => Some (f a)
    | None => None
    end.

#[export] Instance Return_option : Return option :=
  fun A => Some.

#[export] Instance Join_option : Join option :=
  fun A (m : option (option A)) =>
    match m with
    | Some m' => m'
    | None => None
    end.
